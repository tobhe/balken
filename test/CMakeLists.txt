set(Boost_USE_STATIC_LIBS OFF)
set(Boost_USE_MULTITHREADED ON)
set(Boost_USE_STATIC_RUNTIME OFF)
find_package(Boost COMPONENTS gil)

if(Boost_FOUND)

    test.cc
    )
  target_include_directories(test
    ${Boost_INCLUDE_DIRS}

    ${Boost_LIBRARIES}
